The Vehicle is an autonomous agent which will be defined and used accordingly to class Vehicle. The class consists of methods: updating parameteres of Vehicle, applying new vectors(forces) to Vehicle, display agent with vectors on canvas and listed on the left steering behaviours. Below you can see basic methods that will be constantly used in simulations, so they won't appear on other pages. 
class Vehicle {
  constructor(x, y) {
    this.acceleration = new Vector(0, 0),
    this.velocity = new Vector(0, 0),
    this.position = new Vector(x, y),
    this.radius = 30,
    this.maxspeed = 4,
    this.maxforce = 0.1;
    this.desiredVelocity = new Vector(0, 0);
  }
  update() {
    this.velocity = this.velocity.add(this.acceleration);
    this.velocity.limit(this.maxspeed);
    this.position = this.position.add(this.velocity);
    this.acceleration = this.acceleration.multiply(0);
  }
  applyForce(force) {
    this.acceleration = this.acceleration.add(force);
  }
}
