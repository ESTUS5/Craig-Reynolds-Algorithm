  The Flow Field Following is a method that requires implementation of an object which divides space into tiles. Every tile has its own vector. When the vehicle is currently positioned on a certain tile, this tile will return its vector, which will be a new target for the Seek method of the agent.
  The orange vector pointing out of the center of Vehicle is a visualization of the desired velocity and the violet one shows the current velocity vector. The green vector is a flow of tile that the vehicle is occupying right now.
  In order to change the vector of a tile use the left mouse button in tile's space. This action will move vector to the direction of your cursor.

class Board{
 tile(obj_posX, obj_posY) {
  for (let x = 0; x < this.divisions; x++) {
   if (obj_posX < this.vectors_arr[x][0][2]) {
    for (let y = 0; y < this.divisions; y++) {
     if (obj_posY < this.vectors_arr[x][y][3]) {
      if (this.vectors_arr[x][y][0] > (this.vectors_arr[x][y][2]
       - (canvas.width/(this.divisions*2)))) {
        var xb = this.vectors_arr[x][y][2] + 1;
      }
      else {
         var xb = this.vectors_arr[x][y][2]
             - (canvas.width/this.divisions) - 1;
      }
      var a = (this.vectors_arr[x][y][1] - (this.vectors_arr[x][y][3]
       - (canvas.height/(this.divisions*2)))) / (this.vectors_arr[x][y][0]
        - (this.vectors_arr[x][y][2] - (canvas.width/(this.divisions*2))));
      if (a == Infinity || a == -Infinity) {
        a = 0;
      }
      var b = obj_posY - a * obj_posX;
      var yb = a * xb + b;
      if (yb > this.vectors_arr[x][y][3]) {
        yb = this.vectors_arr[x][y][3];
        xb = (yb - b) / a;
      }
      else if (yb < this.vectors_arr[x][y][3]
         - (canvas.height/this.divisions)) {
        yb = this.vectors_arr[x][y][3] - (canvas.height/this.divisions);
        xb = (yb - b) / a;
      }
      return [xb, yb];
     }
    }
  }
}
